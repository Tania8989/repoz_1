#include <iostream>
#include "math.h"
#include <iomanip>
using namespace std;
int main()
{
    int a[10];
    cout << "Введіть елементи масиву: " << endl;
    for(int i = 0; i < 10; i++)
    {
        cout << "a[" << i +1 << "]";
        cin >> a[i];
    }
    cout << "\n Масив: \n";
    for(int i = 0; i < 10; i++)
    cout << setw(5) << a[i] << endl;
    double prod = 1.0;
    int count = 0;
    for(int i = 0; i < 10; i++)
    if(a[i] > 0 && a[i] % 2 == 0){
        prod *= a[i];
        count++;
    }
    if(count > 0){
        double geometricMean = pow(prod, 1.0 / count);
        cout << "Середнє геометричне всіх парних елементів масиву дорівнює: " << geometricMean << endl;
    }
    else{
        cout << "Парні елементи відсутні у масиві" << endl;
    }
}
